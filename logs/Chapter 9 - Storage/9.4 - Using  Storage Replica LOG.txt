PS C:\Foo> # 14. Creating an SR replica group
PS C:\Foo> $SRHT =  @{
             SourceComputerName       = 'SRV1'
             SourceRGName             = 'SRV1RG'
             SourceVolumeName         = 'F:'
             SourceLogVolumeName      = 'G:'
             DestinationComputerName  = 'SRV2'
             DestinationRGName        = 'SRV2RG'
             DestinationVolumeName    = 'F:'
             DestinationLogVolumeName = 'G:'
             LogSizeInBytes           = 2gb
           }
